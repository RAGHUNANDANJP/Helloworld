pipeline {
    agent any

    stages {
        stage('Stage1 clone') {
            steps {
                echo 'git clone the repo'
                git 'https://github.com/RAGHUNANDANJP/Helloworld.git'
            }
        }
        stage('Stage2 Build') {
            steps {
                echo 'Build  the project'
                sh "mvn -Dmaven.test.failure.ignore=true clean package"
            }
        }
        stage('Stage3 Deploy') {
            steps {
                echo 'Deploy the project to Tomcat'
                deploy adapters: [tomcat9(credentialsId: 'tomcat_admin', path: '', url: 'http://52.221.200.199:8080//')], contextPath: null, war: '**/*.war'
            }
        }
        stage('Stage4 Deploy') {
            steps {
                echo 'Deploy the project to test team 1'
            }
        }
        stage('Stage5 Production') {
            steps {
                echo 'Deploy the project to [production]'
            }
        }        
    }
}
